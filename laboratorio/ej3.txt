
(DEFUN DUMP-DB ()
  (DOLIST (CD *DB*)
    (FORMAT T "~{~A:~10T~A~%~}~%" CD)))

(DEFVAR *DB* NIL)

(DEFUN CREAR-CD(TITULO ARTISTA CALIFICACION QUEMADO)
    (LIST :TITULO TITULO :ARTISTA ARTISTA :CALIFICACION CALIFICACION :QUEMADO QUEMADO))

(DEFUN AGREGAR-REGISTRO(CD) (PUSH CD *DB*))

(AGREGAR-REGISTRO (CREAR-CD "DESPUES DE LA PLAYA" "BAD BUNNY" "10/10" T))
(AGREGAR-REGISTRO (CREAR-CD "EL AFRICANO" "WILFREDO VARGAS" "10/10" NIL))

(DEFUN LEER-MOSTRAR(MENSAJE)
    (FORMAT *QUERY-IO*  "~A:" MENSAJE)
    (FORCE-OUTPUT *QUERY-IO*)
    (READ-LINE *QUERY-IO*));ESTA ES UNA FUNCION QUE MUESTRA UN MENSAJE Y PIDE ENTRADA AL USARIO.


(DEFUN PROMPT-FOR-CD ()
  (CREAR-CD
   (LEER-MOSTRAR "TITLE")
   (LEER-MOSTRAR "ARTIST")
   (LEER-MOSTRAR "RATING")
   (LEER-MOSTRAR "RIPPED [Y/N]")))

(PROMPT-FOR-CD)

(DUMP-DB)
